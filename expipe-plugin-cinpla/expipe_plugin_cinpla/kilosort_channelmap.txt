%  create a channel map file

Nchannels = {}; % number of channels
split_probe = {}; % number of channels per group
connected = true(Nchannels, 1);
chanMap   = 1:Nchannels;
chanMap0ind = chanMap - 1;
[xcoords, ycoords] = meshgrid(1:split_probe, 1:Nchannels/split_probe);
xcoords = reshape(xcoords.', 1, [])*50;
ycoords = reshape(ycoords.', 1, [])*50;
kcoords   = zeros(Nchannels, Nchannels/split_probe); % grouping of channels (i.e. tetrode groups)
for i = 1:Nchannels/split_probe
    kcoords(split_probe*(i-1)+1:split_probe*i, i) = 1;
end

fs = {}; % sampling frequency
save(fullfile(fpath, 'chanMap.mat'), ...
    'chanMap','connected', 'xcoords', 'ycoords', 'kcoords', 'chanMap0ind', 'fs')
